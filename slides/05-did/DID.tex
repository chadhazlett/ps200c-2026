\begin{frame}{Weakening or Avoiding Selection on Observables}
\onslide<1-> SOO/CI has become the standard approach across many disciplines \\
\footnotesize
\begin{itemize}
\item<2-> The common formula even now in many fields: ``control for stuff'', say you know there may be confounders, but then treat the result as tentatively correct anyway.
\end{itemize}
\normalsize
\onslide<3->  Much of the focus on causal inference starts by noting the insufficiency of SOO/CI and looking for alternatives. \\
\footnotesize
\begin{itemize}
\item<4-> A (relatively small) number of identification strategies have gained popularity as alternatives: DID, IV, RDD, and some time-series approaches
\item<5-> These identification strategies are often associated with the ``credibility revolution'', but they are not ``more causal'' than SOO/CI -- they just live on different assumptions. 
\item<6-> As with matching, you'll find there is some degree of inappropriate faith in these techniques because of this association
\item<7-> Your job is to see that by understanding the assumptions on which these approaches rest. 
\end{itemize}
\end{frame}

\begin{frame}{DID in the Time of Cholera}
\small
\begin{itemize}
\item Prevailing 19th century wisdom was that cholera was spread by ``bad air''
\smallskip 
\item John Snow, a London physician thought otherwise
\smallskip
\item In 1849, the Southwark and Vauxhall water company and the Lambeth company got water from the same polluted part of the Thames in central London. Snow assessed cholera mortality in both areas. \smallskip
\item In 1852, Lambeth moved its waterworks up stream.
\smallskip
\item Shortly after, cholera death rates fell sharply in areas served by Lambeth, but not area served by S \& V. 
\end{itemize}
\bigskip
\onslide<2-> Intuitions:
\begin{itemize}
\item<3-> Why not just compare the two in 1852?
\item<4-> Why not just compare Lambeth's service area post-minus-pre? 
\end{itemize}
\end{frame}

\begin{frame}
  \frametitle{Two Groups, Two Periods}
\begin{Definition}
Two groups:
\footnotesize
\begin{itemize}
\item $D=1$ for (eventually) treated units
\item $D=0$ for control units
\end{itemize}\medskip
\small
Two periods:
\footnotesize
\begin{itemize}
\item $T=0$ Pre-Treatment period
\item $T=1$ Post-Treatment period
\end{itemize}\medskip
\small
Potential outcomes $Y_{di}(t)$
\footnotesize
\begin{itemize}
\item $Y_{1i}(t)$ potential outcome unit $i$ attains in period $t$ when treated between $t$ and $t-1$
\item $Y_{0i}(t)$ potential outcome unit $i$ attains in period $t$ when non-treated
\end{itemize}
\end{Definition}
\end{frame}

\begin{frame}
  \frametitle{Two Groups and Two Periods}
\begin{Definition}
Causal effect for unit $i$ at time $t$ is
\footnotesize
$$\tau_{it}=Y_{1i}(t)-Y_{0i}(t)$$
\small
\medskip
Observed outcomes $Y_i(t)$ are realized as
\footnotesize
$$Y_i(t)=Y_{1i}(t)\cdot (D_i*(t=1))+Y_{0i}(t)\cdot (1-(D_i*(t=1))$$
\end{Definition}
%\medskip
%\onslide<2-> For $D$ that occurs after $t=0$, we have:
%\small
%$$Y_i(t)=Y_{1i}(t)\cdot D_i(t)+Y_{0i}(t)\cdot (1-D_i(t))$$
\small
\begin{estm}[ATT post-treatment]
$$\tau_{ATT}=\E[Y_{1i}(1)-Y_{0i}(1)|D_i=1]$$
\end{estm}
\end{frame}

\begin{frame}
  \frametitle{Two Groups and Two Periods}  
  \begin{estm}[ATT]
$$\tau_{ATT}=\E[Y_{1i}(1)|D_i=1]-\textcolor{Red}{\E[Y_{0i}(1)|D_i=1]}$$
\end{estm}
\onslide<2-> Observed Moments:
\small
\begin{center}
\begin{tabular}{ccc}
    {\bf } & Post-Period (T=1) & Pre-Period (T=0) \\
\hline \hline
\multirow{2}{*}{Treated D=1}  & \multirow{2}{*}{$\E[Y_1(1)|D=1]$} & \multirow{2}{*}{$\E[Y_0(0)|D=1]$}  \\
 & &  \\
\hline
\multirow{2}{*}{Control D=0}  & \multirow{2}{*}{$\E[Y_0(1)|D=0]$} & \multirow{2}{*}{$\E[Y_0(0)|D=0]$}  \\
 & &  \\
\hline
\end{tabular}
\end{center}
\begin{overprint}
\onslide<3|handout:3>
\begin{problem}
\footnotesize
Missing potential outcome: \textcolor{red}{$\E[Y_0(1)|D=1]$}, ie. what is the average non-treatment outcome in the post-treatment period, for the treated?
\end{problem}
\onslide<4|handout:4>
ID Strategy 1 (not DID): Over-time (``Post-minus-Pre'') Comparison
\begin{itemize}
\item Estimand: $\E[Y_1(1)|D=1]-\E[Y_0(0)|D=1]$
\item Works if: $\E[Y_0(0)|D=1]=\textcolor{red}{\E[Y_0(1)|D=1]}$
\item What does this assumption mean?
\end{itemize}
\onslide<5|handout:5>
ID Strategy 2 (not DID): Cross-Sectional Comparison (in Post) \begin{itemize}
\item Use: $\E[Y(1)|D=1]-\E[Y(1)|D=0]$
\item Assumes: $\E[Y_0(1)|D=0]=\textcolor{red}{\E[Y_0(1)|D=1]}$
\item What does this mean?
\end{itemize}
\onslide<6|handout:6>
ID Strategy 3: Difference-in-Differences (DID)\begin{itemize}\small
\item Use: \footnotesize $$\E[Y(1)-Y(0)|D=1]- \E[Y(1)-Y(0)|D=0]$$
\item \small Assumes: \footnotesize $\E[\textcolor{Red}{Y_0(1)}-Y_0(0)|D=1]=\E[Y_0(1)-Y_0(0)|D=0]$
\begin{center}
\small \textcolor{Blue}{``Parallel trends''} assumption
\end{center}
\end{itemize}
\end{overprint}
\end{frame}

\begin{frame}
  \frametitle{Identification with Difference-in-Differences}
\small
\vspace{-.2in}
\begin{iass}[Parallel Trends]
Conventionally we write,
$$\textcolor{Red}{\E[Y_0(1)-Y_0(0)|D=1]}=\textcolor{Blue}{\E[Y_0(1)-Y_0(0)|D=0]}$$
But note that this is equivalent to:  
\[ \Delta Y_0 \perp D   \]
where $\Delta Y_0 \equiv \Y_0(1)-\Y_0(0)$
\end{iass}
\bigskip
\onslide<2>
\begin{ires}
Under parallel trends, the ATT is identified by the DID estimator:
\vspace{-.1in}
\begin{align*}
\tau_{DID} &=  \{\E[Y(1)|D=1]-\E[Y(0)|D=1]\}-\{\E[Y(1)|D=0]-\E[Y(0)|D=0]\} \\
&=\E[Y_1(1)-Y_0(1)|D=1]=ATT
\end{align*}
\end{ires}
\end{frame}

\begin{frame}{Proof}
\footnotesize
\begin{align*}
\onslide<1->{\tau_{DID} &= \{\E[Y(1)|D=1]-\E[Y(0)|D=1]\}-\{\E[Y(1)|D=0]-\E[Y(0)|D=0]\}} \\
\onslide<2->{&= \{\E[Y_1(1)|D=1]-\E[Y_0(0)|D=1]\}-\textcolor{Blue}{\{\E[Y_0(1)|D=0]-\E[Y_0(0)|D=0]\}}}\\
\onslide<3->{&= \{\E[Y_1(1)|D=1]-\E[Y_0(0)|D=1]\}-\textcolor{Red}{\{\E[Y_0(1)|D=1]-\E[Y_0(0)|D=1]\}}}\\
\onslide<4->{&= \E[Y_1(1)|D=1]-\textcolor{Red}{\E[Y_0(1)|D=1]}}\\
\onslide<5->{&= ATT}
\end{align*}
\small \\
\bigskip
\onslide<6-> Another view: additive effects
\footnotesize
\begin{itemize}
\item<7-> Change in control unit's outcome, $\Delta Y_0$, is ``effect of time'', $\Delta Y_0=\delta_0$
\item<8-> Change in treated unit's outcome is effect of time and treatment, $\Delta Y_1=\delta_1+\tau_{ATT}$
\item<9-> We estimate $DID = \Delta Y_1-\Delta Y_0 = (\delta_1+ \tau_{ATT}) -\delta_0 $
\item<10-> ...equals $\tau_{ATT}$ if $\delta_0=\delta_1$ 
\end{itemize}
\end{frame}

\begin{align*}
\tau_{DID} &= \{\E[Y(1)|D=1]-\E[Y(1)|D=0]\}-\{\E[Y(0)|D=1]-\E[Y(0)|D=0]\}
\end{align*}
\end{estm}
\bigskip

\begin{esti}[Regression: Repeated Cross-Sections]\small
\textbf{In the two period case}, the same estimator can be computed by regression:
\[
Y = \mu + \gamma\cdot D + \delta\cdot T + \tau \cdot (D\cdot T) + \varepsilon,
\]
where $\E[\varepsilon|D,T]=0$.
\end{esti}\medskip

\begin{frame}{Other approaches to worrying about parallel trends}
\begin{enumerate}
\item \textbf{Sensitivity/ partial ID}\bigskip
\small
\begin{itemize}
\item Recently-best-known is perhaps Rambachan \& Roth (2023)

\begin{frame}
  \frametitle{Difference-in-Differences: Further Threats to Validity}
\begin{enumerate}
\setcounter{enumi}{3}
 \item<1-> \textbf{Functional form dependence}:  The differencing is not-invariant to changes of scale. Magnitude or sign of the DID effect may be sensitive to this.
 \begin{itemize}
   \item<2-> Higher risk when average outcomes for controls and treated are very different at baseline.
   \medskip
   
 \item<3-> Example: Cash transfers for refugees
 \begin{itemize}
   \item<4-> income of those receiving vouchers rises from 5 to 10
   \item<5-> over same time, income for comparably skilled group in host community (no vouchers) changes from 20 to 25
   \item<6-> on original scale, $DID = (10 - 5) - (25 - 20) = 0\%$ effect of voucher \medskip \pause
   \item<7-> but in terms of percentage changes or logs, \\
   $[100]-[25]= 75$ percentage points \\
   $[log(10) - log(5)] - [log(25) - log(20)] = log(2)- log(1.25)  = .47$
 \end{itemize}\bigskip
 \item<8-> In general, more similarity in original outcomes is better both for this reason and to make parallel trends assumption more believable
 \end{itemize}
\end{enumerate}
\end{frame}

\begin{frame}
  \frametitle{Matching and difference-in-differences}
\begin{itemize}
\small
\item<1-> Parallel trends is more plausible among units that are more similar (both on $X$ and pre-tx $Y$). One strategy: \onslide<2-> \textcolor{Blue}{combine matching (or weighting) and DID}\\
\medskip
\begin{itemize}
  \item<3-> Match on pre-treatment covariates and (lagged) outcomes\medskip
  \item<4-> Run difference-in-differences regression in matched data-set\medskip
  \item<5-> Can also use inverse-propensity score weighting (Hirano, Imbens, and Ridder 2003; Imai and Kim 2012)\medskip
  \item<6-> (If you match/weight so that the treated and control groups have equal means on pre-treatment Y, then DID is just Diff.) 
\end{itemize}\bigskip
%\item This strategy is more widely used now:\medskip

%%  \item Lenz and Ladd (2009). Exploiting a Rare Communication Shift to Document the Persuasive Power of the News Media, \emph{American Journal of Political Science}, 53(2): 394-410.
%\end{itemize} \bigskip
\item<6-> Randomized experiment, natural experiment, or a  discontinuity can improve parallel trends assumption by asserting ``$Y(0)$-trend'' is balanced. So DID can be used in combination.
\item<7-> This is just to say you can use the change-score rather than level of $Y$ as your outcome in an experiment.\item<7-> With few treated units and/or many pre-treatment periods: synthetic control and related methods are applicable, but different assumptions.
\end{itemize}
\end{frame}


\begin{frame}{``Modern'' DID?}
\small
\begin{itemize}
\item<1-> Everything we've done here is to give you a foundation in ``classic'' DID, meaning two groups, two periods. 
\item<2-> Sometimes you have different units moving into treatment at different times and you still want to imagine something like parallel trends.
\item<3-> People long used ``two-way fixed effect'' models in these cases and unfortunately still called this DID (?!)
\item<4-> Then they discovered TWFE does not have a straightforward interepretation akin to DID and is troubled by either biases or confusing interpretations due to:
\begin{itemize}
\footnotesize
\item<5-> Treatment effect heterogeneity
\item<6-> Already-treated as controls
\item<7-> Event time misalignment	
\item<8-> Global vs. local DID assumptions 
\end{itemize}
\end{itemize}
\end{frame}


\begin{frame}{What to do with staggered treatments etc.?}
\small
\onslide<1-> You can always run a series of separate well-constructed DID for each wave of treatment adoption, understand what each says, then average later.\\
\bigskip

\begin{frame}{Example of Matching-plus-DID: Ladd and Lenz (2009)}

\small
\begin{quote}
Using panel data and matching techniques, we exploit a rare change in communication flows -- the endorsement switch to the Labour Party by several prominent British newspapers before the 1997 United Kingdom general election—to study the persuasive power of the news media. These unusual endorsement switches provide an opportunity to test for news media persuasion while avoiding methodological pitfalls that have plagued previous studies. By comparing readers of newspapers that switched endorsements to similar individuals who did not read these newspapers, we estimate that these
papers persuaded a considerable share of their readers to vote for Labour. %Depending on the statistical approach, the point estimates vary from about 10\% to as high as 25\% of readers. These findings provide rare evidence that the news media exert a powerful influence on mass political behavior.
\end{quote}
\end{frame}
